1. Factorial (n!)Easy
basic recursion
Algorithm: Base case n==0 or n==1 → return 1. Recursive case: n * factorial(n-1). Each call reduces n by 1 until base case.
public class Factorial {
    public static int factorial(int n) {
        if (n == 0 || n == 1) return 1; // base case
        return n * factorial(n - 1);           // recursive case
    }
    public static void main(String[] args) {
        System.out.println(factorial(5)); // 120
    }
}
Time O(n)
Space O(n) stack
2. Fibonacci number (LeetCode 509)Easy
tree recursion
Algorithm: Base cases: fib(0)=0, fib(1)=1. Recursive: fib(n-1)+fib(n-2). Two branches at every node — exponential without memoization. With memo: O(n).
public class Fibonacci {
    private int[] memo;

    public int fib(int n) {
        memo = new int[n + 2];
        Arrays.fill(memo, -1);
        return solve(n);
    }

    private int solve(int n) {
        if (n == 0) return 0;
        if (n == 1) return 1;
        if (memo[n] != -1) return memo[n];
        memo[n] = solve(n - 1) + solve(n - 2);
        return memo[n];
    }
}
Time O(n) with memo
Space O(n)
3. Pow(x, n) — LeetCode 50Easy
divide & conquer
Algorithm: Fast power. If n is even: pow(x,n) = half*half where half=pow(x,n/2). If odd: x*half*half. Halves the problem each call → O(log n).
public class MyPow {
    public double myPow(double x, int n) {
        long N = n; // handle Integer.MIN_VALUE
        if (N < 0) { x = 1 / x; N = -N; }
        return fastPow(x, N);
    }

    private double fastPow(double x, long n) {
        if (n == 0) return 1.0; // base case
        double half = fastPow(x, n / 2);
        if (n % 2 == 0) return half * half;
        else             return x * half * half;
    }
}
Time O(log n)
Space O(log n)

4. Reverse string — LeetCode 344Easy
two-pointer recursion
Algorithm: Swap s[left] and s[right]. Recurse with left+1, right-1. Base: left >= right → stop.
public class ReverseString {
    public void reverseString(char[] s) {
        reverse(s, 0, s.length - 1);
    }

    private void reverse(char[] s, int l, int r) {
        if (l >= r) return; // base case
        char tmp = s[l]; s[l] = s[r]; s[r] = tmp;
        reverse(s, l + 1, r - 1);
    }
}
Time O(n)
Space O(n/2) stack
5. Valid palindrome check (recursive)Easy
two-pointer recursion
Algorithm: Check s[left]==s[right]. If mismatch → false. Else recurse inward. Base: left>=right → true (all characters matched).
public class PalindromeCheck {
    public boolean isPalindrome(String s) {
        String clean = s.toLowerCase()
            .replaceAll("[^a-z0-9]", "");
        return check(clean, 0, clean.length() - 1);
    }

    private boolean check(String s, int l, int r) {
        if (l >= r) return true;
        if (s.charAt(l) != s.charAt(r)) return false;
        return check(s, l + 1, r - 1);
    }
}
Time O(n)
Space O(n)

7. Binary search — LeetCode 704Easy
divide & conquer
Algorithm: Find mid. If a[mid]==target return mid. If target < a[mid] search left half. Else search right half. Base: lo > hi → -1.
public class BinarySearch {
    public int search(int[] nums, int target) {
        return bs(nums, 0, nums.length - 1, target);
    }

    private int bs(int[] a, int lo, int hi, int t) {
        if (lo > hi) return -1;
        int mid = lo + (hi - lo) / 2;
        if      (a[mid] == t) return mid;
        else if (a[mid] > t)  return bs(a, lo, mid-1, t);
        else                   return bs(a, mid+1, hi, t);
    }
}
Time O(log n)
Space O(log n)
8. First bad version — LeetCode 278Easy
binary search recursion
Algorithm: Same as binary search. If isBadVersion(mid) → bad is at mid or left. Else → bad is to the right. Narrow recursively.
public class FirstBadVersion {
    public int firstBadVersion(int n) {
        return find(1, n);
    }

    private int find(int lo, int hi) {
        if (lo == hi) return lo; // base: narrowed to one
        int mid = lo + (hi - lo) / 2;
        if (isBadVersion(mid)) return find(lo, mid);
        else                    return find(mid + 1, hi);
    }
}
Time O(log n)
Space O(log n)

9. Reverse linked list — LeetCode 206Easy
linked list recursion
Algorithm: Base: node is null or last → return it. Recurse to end. On unwind: make next.next point back to current, set current.next = null. Returns the new head (last node).
public class ReverseList {
    public ListNode reverseList(ListNode head) {
        // base: empty or single node
        if (head == null || head.next == null)
            return head;
        ListNode newHead = reverseList(head.next);
        head.next.next = head; // reverse the link
        head.next = null;      // cut original link
        return newHead;        // propagate new head up
    }
}
Time O(n)
Space O(n) stack
10. Merge two sorted lists — LeetCode 21Easy
linked list recursion
Algorithm: Compare heads. Pick smaller → its next = merge(its next, other list). Base: either list is null → return the other.
public class MergeLists {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null) return l2; // base cases
        if (l2 == null) return l1;
        if (l1.val <= l2.val) {
            l1.next = mergeTwoLists(l1.next, l2);
            return l1;
        } else {
            l2.next = mergeTwoLists(l1, l2.next);
            return l2;
        }
    }
}
Time O(m+n)
Space O(m+n)

11. Max depth of binary tree — LeetCode 104Easy
tree DFS recursion
Algorithm: Depth = 1 + max(depth(left), depth(right)). Base: null → 0. Each node adds 1 to the deeper subtree depth.
public class MaxDepth {
    public int maxDepth(TreeNode root) {
        if (root == null) return 0; // base case
        int leftDepth  = maxDepth(root.left);
        int rightDepth = maxDepth(root.right);
        return 1 + Math.max(leftDepth, rightDepth);
    }
}
Time O(n)
Space O(h) stack
12. Same tree — LeetCode 100Easy
tree DFS recursion
Algorithm: Both null → true. One null → false. Values differ → false. Else recurse on left subtrees AND right subtrees.
public class SameTree {
    public boolean isSameTree(TreeNode p, TreeNode q) {
        if (p == null && q == null) return true;
        if (p == null || q == null) return false;
        if (p.val != q.val)           return false;
        return isSameTree(p.left, q.left) &&
               isSameTree(p.right, q.right);
    }
}
Time O(n)

13. Symmetric tree — LeetCode 101Easy
tree DFS recursion
Algorithm: Check if left subtree is a mirror of right subtree. Mirror: left.val==right.val AND left.left mirrors right.right AND left.right mirrors right.left.
public class SymmetricTree {
    public boolean isSymmetric(TreeNode root) {
        return isMirror(root.left, root.right);
    }

    private boolean isMirror(TreeNode l, TreeNode r) {
        if (l == null && r == null) return true;
        if (l == null || r == null) return false;
        return (l.val == r.val)
            && isMirror(l.left, r.right)
            && isMirror(l.right, r.left);
    }
}
Time O(n)
Space O(h)
14. Path sum — LeetCode 112Easy
tree DFS recursion
Algorithm: Subtract node value from target. At leaf (no children): check if remaining==0. At non-leaf: recurse left or right.
public class PathSum {
    public boolean hasPathSum(TreeNode root, int target) {
        if (root == null) return false;
        target -= root.val;
        // leaf check
        if (root.left == null && root.right == null)
            return target == 0;
        return hasPathSum(root.left, target) ||
               hasPathSum(root.right, target);
    }
}
Time O(n)
Space O(h)
15. Invert binary tree — LeetCode 226Easy
tree DFS recursion
Algorithm: Swap left and right children. Recurse on both. Base: null → return null.
public class InvertTree {
    public TreeNode invertTree(TreeNode root) {
        if (root == null) return null;
        TreeNode tmp  = root.left;
        root.left     = root.right;
        root.right    = tmp;
        invertTree(root.left);
        invertTree(root.right);
        return root;
    }
}

16. Binary tree traversals — LeetCode 144/94/145Easy
tree DFS traversal
Algorithm: Pre: root→left→right. In: left→root→right. Post: left→right→root. Each is just reordering the three recursive calls.
public class Traversals {
    List<Integer> res = new ArrayList<>();

    void preorder(TreeNode n) {
        if(n==null) return;
        res.add(n.val);        // ROOT first
        preorder(n.left);
        preorder(n.right);
    }

    void inorder(TreeNode n) {
        if(n==null) return;
        inorder(n.left);
        res.add(n.val);        // ROOT middle
        inorder(n.right);
    }

    void postorder(TreeNode n) {
        if(n==null) return;
        postorder(n.left);
        postorder(n.right);
        res.add(n.val);        // ROOT last
    }
}
Time O(n)
Space O(h)

17. Subsets — LeetCode 78Easy
backtracking / include-exclude
Algorithm: At each index, make two choices: include nums[i] in current subset OR skip it. Recurse on next index. When index==n, add current subset to result.
public class Subsets {
    List<List<Integer>> res = new ArrayList<>();

    public List<List<Integer>> subsets(int[] nums) {
        backtrack(nums, 0, new ArrayList<>());
        return res;
    }

    void backtrack(int[] nums, int idx, List<Integer> cur) {
        res.add(new ArrayList<>(cur)); // add snapshot
        for (int i = idx; i < nums.length; i++) {
            cur.add(nums[i]);           // include
            backtrack(nums, i + 1, cur);
            cur.remove(cur.size() - 1); // undo
        }
    }
}
Time O(2^n)
Space O(n)
18. Permutations — LeetCode 46Easy
backtracking
Algorithm: For each unused number, add it to current, mark used, recurse. When current size equals n, add to result. Undo: remove last, unmark.
public class Permutations {
    List<List<Integer>> res = new ArrayList<>();
    boolean[] used;

    public List<List<Integer>> permute(int[] nums) {
        used = new boolean[nums.length];
        backtrack(nums, new ArrayList<>());
        return res;
    }

    void backtrack(int[] nums, List<Integer> cur) {
        if (cur.size() == nums.length) {
            res.add(new ArrayList<>(cur)); return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            used[i] = true;
            cur.add(nums[i]);
            backtrack(nums, cur);
            cur.remove(cur.size() - 1);
            used[i] = false;
        }
    }
}
Time O(n! × n)
Space O(n)

19. Climbing stairs — LeetCode 70Easy
recursion + memoization
Algorithm: From stair n, you came from n-1 or n-2. So ways(n) = ways(n-1) + ways(n-2). Exactly Fibonacci. Base: ways(0)=1, ways(1)=1.
public class ClimbStairs {
    int[] memo;
    public int climbStairs(int n) {
        memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return ways(n);
    }
    int ways(int n) {
        if (n <= 1) return 1;
        if (memo[n] != -1) return memo[n];
        return memo[n] = ways(n-1) + ways(n-2);
    }
}
Time O(n)
Space O(n)

21. Balanced binary tree — LeetCode 110Easy
tree DFS recursion
Algorithm: For each node, get heights of left and right subtrees. If |left-right| > 1 → not balanced. Recurse both sides. Return -1 as sentinel for unbalanced.
public class BalancedTree {
    public boolean isBalanced(TreeNode root) {
        return height(root) != -1;
    }
    int height(TreeNode node) {
        if (node == null) return 0;
        int lh = height(node.left);
        if (lh == -1) return -1; // propagate unbalanced
        int rh = height(node.right);
        if (rh == -1) return -1;
        if (Math.abs(lh - rh) > 1) return -1;
        return 1 + Math.max(lh, rh);
    }
}
Time O(n)
Space O(h)
22. Diameter of binary tree — LeetCode 543Easy
tree DFS recursion
Algorithm: Diameter through a node = leftHeight + rightHeight. Track global max. Return 1+max(lh,rh) as height up to parent.
public class DiameterTree {
    int maxDia = 0;
    public int diameterOfBinaryTree(TreeNode root) {
        depth(root);
        return maxDia;
    }
    int depth(TreeNode node) {
        if (node == null) return 0;
        int lh = depth(node.left);
        int rh = depth(node.right);
        maxDia = Math.max(maxDia, lh + rh); // path through node
        return 1 + Math.max(lh, rh);
    }
}
Time O(n)
Space O(h)
23. Merge two binary trees — LeetCode 617Easy
tree DFS recursion
Algorithm: If one node is null, return the other. Else add values, recurse on both left and right children.
public class MergeTrees {
    public TreeNode mergeTrees(TreeNode r1, TreeNode r2) {
        if (r1 == null) return r2;
        if (r2 == null) return r1;
        r1.val += r2.val; // merge values
        r1.left  = mergeTrees(r1.left,  r2.left);
        r1.right = mergeTrees(r1.right, r2.right);
        return r1;
    }
}

26. Happy number — LeetCode 202Easy
recursion + set (cycle detection)
Algorithm: Sum of squares of digits. If result==1 → happy. If seen before → cycle → not happy. Else recurse on new sum.
public class HappyNumber {
    Set<Integer> seen = new HashSet<>();

    public boolean isHappy(int n) {
        if (n == 1)         return true;
        if (seen.contains(n)) return false; // cycle
        seen.add(n);
        int next = 0;
        while (n > 0) {
            int d = n % 10;
            next += d * d;
            n /= 10;
        }
        return isHappy(next);
    }
}
Time O(log n)
Space O(log n)

27. Remove all adjacent duplicates — LeetCode 1047Easy
recursion with stack-like thinking
Algorithm: Build result using a char array (stack). Scan each char: if top of stack equals current char, pop (remove adjacent pair). Else push. O(n) one pass.
public class RemoveDuplicates {
    public String removeDuplicates(String s) {
        char[] stack = new char[s.length()];
        int top = -1;
        for (char c : s.toCharArray()) {
            if (top >= 0 && stack[top] == c)
                top--;        // pop (remove pair)
            else
                stack[++top] = c; // push
        }
        return new String(stack, 0, top + 1);
    }
}
Time O(n)
Space O(n)
28. Reverse words in a string (recursive)Easy
string recursion
Algorithm: Find first word, recurse on rest of string. Append current word after result of recursion → words reversed.
public class ReverseWords {
    public String reverseWords(String s) {
        String[] words = s.trim().split("\\s+");
        return rev(words, 0);
    }

    private String rev(String[] w, int i) {
        if (i == w.length - 1) return w[i]; // base
        return rev(w, i + 1) + " " + w[i];
    }
}
Time O(n)
Space O(n)

29. Search in a BST — LeetCode 700Easy
BST recursion
Algorithm: If val==root → return root. If val < root.val → go left. Else go right. Base: null → not found.
public class SearchBST {
    public TreeNode searchBST(TreeNode root, int val) {
        if (root == null)      return null; // not found
        if (root.val == val)   return root; // found
        if (val < root.val)
            return searchBST(root.left, val);
        return searchBST(root.right, val);
    }
}
Time O(h)
Space O(h)
30. Insert into a BST — LeetCode 701Easy
BST recursion
Algorithm: If null → create new node here. If val < root → insert into left subtree. Else insert into right subtree. BST property maintained automatically.
public class InsertBST {
    public TreeNode insertIntoBST(TreeNode root, int val) {
        if (root == null)
            return new TreeNode(val); // insert here
        if (val < root.val)
            root.left  = insertIntoBST(root.left, val);
        else
            root.right = insertIntoBST(root.right, val);
        return root;
    }
}
Time O(h)
Space O(h)
