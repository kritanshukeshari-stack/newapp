EasyMax, min, sum, average
Algorithm: Single pass through array. Maintain running max, min, sum. Average = sum / n.
import java.util.*;
public class MaxMinSum {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();

        int max = a[0], min = a[0];
        long sum = 0;
        for (int x : a) {
            if (x > max) max = x;
            if (x < min) min = x;
            sum += x;
        }
        double avg = (double) sum / n;
        System.out.println("Max: " + max);
        System.out.println("Min: " + min);
        System.out.println("Sum: " + sum);
        System.out.println("Avg: " + avg);
    }
}

                second = x;
            }
        }

            System.out.println("No second largest");
        else
            System.out.println("Second largest: " + second);
    }

import java.util.*;

            if (freq[i] > 0)
                System.out.println(i + " appears " + freq[i] + " times");
        }
    }
}

import java.util.*;
public class FreqHashMap {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        Map<Integer, Integer> freq = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int x = sc.nextInt();
            freq.put(x, freq.getOrDefault(x, 0) + 1);

                maxFreq = e.getValue();
                mostFreqElem = e.getKey();
            }
            if (e.getValue() < minFreq) {
                minFreq = e.getValue();
                leastFreqElem = e.getKey();
            }
        }
        System.out.println("Most frequent: " + mostFreqElem + " (" + maxFreq + " times)");
        System.out.println("Least frequent: " + leastFreqElem + " (" + minFreq + " times)");
    }
}

EasyRange sum query
Algorithm: Build prefix[i] = prefix[i-1] + a[i-1]. Sum of a[l..r] (0-based) = prefix[r+1] - prefix[l]. Answer multiple queries in O(1) each after O(n) build.
import java.util.*;
public class RangeSum {
    public static void main(String[] args) {

        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();

        long[] prefix = new long[n + 1];
        for (int i = 0; i < n; i++)
            prefix[i + 1] = prefix[i] + a[i];

        int q = sc.nextInt();
        while (q-- > 0) {
            int l = sc.nextInt(), r = sc.nextInt(); // 0-based
            System.out.println(prefix[r + 1] - prefix[l]);
        }
    }
}

EasyEquilibrium index
Algorithm: At index i, left sum = prefix[i], right sum = totalSum - prefix[i+1]. If equal, i is equilibrium. Find all such indices.
import java.util.*;
public class Equilibrium {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();

        long[] pre = new long[n + 1];
        for (int i = 0; i < n; i++) pre[i+1] = pre[i] + a[i];
        long total = pre[n];

        boolean found = false;
        for (int i = 0; i < n; i++) {
            long left = pre[i];
            long right = total - pre[i + 1];
            if (left == right) {
                System.out.println("Equilibrium at index: " + i);
                found = true;
            }
        }
        if (!found) System.out.println("No equilibrium index");
    }
}

BasicMerge two sorted arrays
Algorithm: Use two pointers i, j on each array. Always pick the smaller element and advance that pointer. Fill result array of size n+m.
import java.util.*;
public class MergeSorted {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), m = sc.nextInt();
        int[] a = new int[n], b = new int[m];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();
        for (int i = 0; i < m; i++) b[i] = sc.nextInt();

        int[] result = new int[n + m];
        int i = 0, j = 0, k = 0;
        while (i < n && j < m)
            result[k++] = (a[i] <= b[j]) ? a[i++] : b[j++];
        while (i < n) result[k++] = a[i++];
        while (j < m) result[k++] = b[j++];

        System.out.println(Arrays.toString(result));
    }
}

EasyAnagram check
Algorithm: Two strings are anagrams if their character frequencies match. Build freq[26] for both: increment for s1, decrement for s2. All zeros = anagram.
import java.util.*;
public class AnagramCheck {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String a = sc.next().toLowerCase();
        String b = sc.next().toLowerCase();

        if (a.length() != b.length()) {
            System.out.println("Not anagram");
            return;
        }
        int[] freq = new int[26];
        for (int i = 0; i < a.length(); i++) {
            freq[a.charAt(i) - 'a']++;
            freq[b.charAt(i) - 'a']--;
        }
        for (int f : freq) {
            if (f != 0) {
                System.out.println("Not anagram");
                return;
            }
        }
        System.out.println("Anagram");
    }
}

public class StringOps {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println(s.toUpperCase());

                cnt++; i++;

        }
        System.out.println("Compressed: " + sb);

}

BasicBinary search — exact match
Algorithm: Array must be sorted. Compare target with middle element. If smaller, search left half; if larger, search right half. Repeat until found or range is empty.
import java.util.*;
public class BinarySearch {
    static int binarySearch(int[] a, int target) {
        int lo = 0, hi = a.length - 1;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (a[mid] == target) return mid;
            else if (a[mid] < target) lo = mid + 1;
            else hi = mid - 1;
        }
        return -1;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), target = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();
        Arrays.sort(a);
        int idx = binarySearch(a, target);
        System.out.println(idx == -1 ? "Not found" : "Found at: " + idx);
    }
}

BasicBinary search — lower and upper bound
Algorithm: Lower bound = first index where a[i] >= target. Upper bound = first index where a[i] > target. Both use binary search. Count of target = upperBound - lowerBound.
import java.util.*;
public class LowerUpperBound {
    static int lowerBound(int[] a, int x) {
        int lo = 0, hi = a.length;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (a[mid] < x) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
    static int upperBound(int[] a, int x) {
        int lo = 0, hi = a.length;
        while (lo < hi) {
            int mid = (lo + hi) / 2;
            if (a[mid] <= x) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), x = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();
        Arrays.sort(a);
        int lb = lowerBound(a, x);
        int ub = upperBound(a, x);
        System.out.println("Lower bound index: " + lb);

Algorithm: Use a HashMap to track element frequencies in current window. Size of map = distinct count. Slide window: add right element, remove left element (delete from map if freq drops to 0).
import java.util.*;
public class DistinctInWindow {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Map<Integer, Integer> freq = new HashMap<>();
        for (int i = 0; i < k; i++)
            freq.put(a[i], freq.getOrDefault(a[i], 0) + 1);
        System.out.print("Window [0,"+(k-1)+"] distinct: "+freq.size()+"\n");

        for (int i = k; i < n; i++) {
            freq.put(a[i], freq.getOrDefault(a[i], 0) + 1);
            int outgoing = a[i - k];
            freq.put(outgoing, freq.get(outgoing) - 1);
            if (freq.get(outgoing) == 0) freq.remove(outgoing);
            System.out.print("Window ["+(i-k+1)+","+i+"] distinct: "+freq.size()+"\n");
        }
    }
}

BasicLongest substring without repeating characters
Algorithm: Use a HashSet to track characters in current window. If a[r] already in set, shrink from left until removed. Expand right each step. Track maximum window size.
import java.util.*;
public class LongestUniqueSubstring {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = sc.next();
        int n = s.length();
        Set<Character> window = new HashSet<>();
        int l = 0, maxLen = 0, bestL = 0;

        for (int r = 0; r < n; r++) {
            while (window.contains(s.charAt(r)))
                window.remove(s.charAt(l++));
            window.add(s.charAt(r));
            if (r - l + 1 > maxLen) {
                maxLen = r - l + 1;
                bestL = l;
            }
        }
        System.out.println("Max length: " + maxLen);
        System.out.println("Substring: " + s.substring(bestL, bestL + maxLen));
    }
}
