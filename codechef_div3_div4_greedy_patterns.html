Easy1. Pick largest K elements to maximize sum
Algorithm: Sort descending. Sum first K elements. The greedy choice is always to pick the largest available element first.
import java.util.*;
import java.util.Scanner;

public class MaxSumK {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), k = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();

        Arrays.sort(a);   // ascending
        long sum = 0;
        // pick last k (largest)
        for (int i = n - k; i < n; i++) sum += a[i];

        System.out.println(sum);
    }
}
Time O(n log n)
Space O(1)

Med1. Merge overlapping intervals
Algorithm: Sort by start. Walk through: if current start ≤ last end, merge by extending end. Otherwise push new interval.
import java.util.*;

public class MergeIntervals {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] intervals = new int[n][2];
        for (int i = 0; i < n; i++) {
            intervals[i][0] = sc.nextInt();
            intervals[i][1] = sc.nextInt();
        }
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);

        List<int[]> res = new ArrayList<>();
        for (int[] iv : intervals) {
            if (res.isEmpty() ||
                res.get(res.size()-1)[1] < iv[0]) {
                res.add(new int[]{iv[0], iv[1]});
            } else {
                res.get(res.size()-1)[1] =
                    Math.max(res.get(res.size()-1)[1], iv[1]);
            }
        }
        System.out.println("Merged count: " + res.size());
        for (int[] r : res)
            System.out.println(r[0] + " " + r[1]);
    }
}
Time O(n log n)
Space O(n)
Med2. Minimum number of meeting rooms needed
Algorithm: Sort start times. Use a min-heap of end times. For each meeting, if its start ≥ heap top, reuse room (poll). Otherwise add room. Heap size = rooms needed.
import java.util.*;

public class MeetingRooms {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] meet = new int[n][2];
        for (int i = 0; i < n; i++) {
            meet[i][0] = sc.nextInt();
            meet[i][1] = sc.nextInt();
        }
        Arrays.sort(meet, (a, b) -> a[0] - b[0]);

        PriorityQueue<Integer> heap =
            new PriorityQueue<>(); // min-heap of end times
        for (int[] m : meet) {
            if (!heap.isEmpty() && heap.peek() <= m[0])
                heap.poll(); // reuse room
            heap.offer(m[1]);
        }
        System.out.println("Rooms needed: " + heap.size());
    }
}
Time O(n log n)
Space O(n)
Easy1. Minimum coins to make amount (standard denominations)
Algorithm: Sort coins descending. Greedily use the largest coin that fits. Subtract from amount, repeat. Works only for standard coin systems (not arbitrary denominations — use DP for those).
import java.util.*;

public class MinCoins {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int[] coins = {1, 2, 5, 10, 20, 50, 100, 500, 2000};
        int amount = sc.nextInt();
        int count = 0;

        for (int i = coins.length - 1; i >= 0 && amount > 0; i--) {
            if (coins[i] <= amount) {
                int use = amount / coins[i];
                count += use;
                amount -= use * coins[i];
            }
        }
        System.out.println("Minimum coins: " + count);
    }
}
Time O(amount)
Space O(1)
Note: for arbitrary coin sets, greedy fails — use DP. Div 3/4 always gives canonical coins.

Med2. Assign tasks to workers — minimize max completion time
Algorithm: Sort tasks descending. Assign each task to the worker with current minimum load (use a min-heap of loads). This balances workload optimally.
import java.util.*;

public class AssignTasks {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), w = sc.nextInt();
        int[] tasks = new int[n];
        for (int i = 0; i < n; i++) tasks[i] = sc.nextInt();
        Arrays.sort(tasks); // sort ascending

        // min-heap: stores current load of each worker
        PriorityQueue<Long> heap = new PriorityQueue<>();
        for (int i = 0; i < w; i++) heap.offer(0L);

        for (int i = n - 1; i >= 0; i--) { // largest task first
            long minLoad = heap.poll();
            heap.offer(minLoad + tasks[i]);
        }
        long maxLoad = 0;
        for (long l : heap) maxLoad = Math.max(maxLoad, l);
        System.out.println("Min max time: " + maxLoad);
    }
}

Easy1. Maximum sum by flipping at most K negatives
Algorithm: Sort by absolute value descending. Flip negatives from largest absolute value first. If K flips remain and all are positive, flip the smallest element (if K is odd).
import java.util.*;

public class MaxSumFlipK {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), k = sc.nextInt();
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = sc.nextInt();

        // sort by absolute value descending
        Integer[] boxed = Arrays.stream(a).boxed().toArray(Integer[]::new);
        Arrays.sort(boxed, (x, y) -> Math.abs(y) - Math.abs(x));

        for (int i = 0; i < n && k > 0; i++) {
            if (boxed[i] < 0) { boxed[i] = -boxed[i]; k--; }
        }
        // if k is still odd, flip the smallest absolute value
        if (k % 2 != 0) boxed[n - 1] = -Math.abs(boxed[n - 1]);

        long sum = 0;
        for (int x : boxed) sum += x;
        System.out.println("Max sum: " + sum);
    }
}
Time O(n log n)
Space O(n)

Easy1. Lexicographically smallest string by swapping adjacent
Algorithm: Repeatedly scan left to right. Find smallest character in remaining suffix, move it to current position by swapping adjacent characters. Repeat for each position.
import java.util.*;

public class LexSmallest {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        char[] s = sc.next().toCharArray();
        int n = s.length;

        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++)
                if (s[j] < s[minIdx]) minIdx = j;

            // bubble minIdx to position i
            while (minIdx > i) {
                char tmp = s[minIdx];
                s[minIdx] = s[minIdx - 1];
                s[minIdx - 1] = tmp;
                minIdx--;
            }
        }
        System.out.println(new String(s));
    }
}

Algorithm: Count frequency of each character. Count how many characters have odd frequency. A palindrome can have at most 1 odd-frequency char. So we need oddCount ≤ K and K ≤ n.

        int k = sc.nextInt();
        int[] freq = new int[26];
        for (char c : s.toCharArray()) freq[c - 'a']++;

        int oddCount = 0;
        for (int f : freq) if (f % 2 != 0) oddCount++;

        boolean possible = (k <= s.length() && oddCount <= k);
        System.out.println(possible ? "YES" : "NO");
    }
}

Med1. Gas station — can you complete the circuit?
Algorithm: If total gas ≥ total cost, solution exists. Track running tank. When tank goes negative, the start must be the next station. Reset tank and start.
import java.util.*;

public class GasStation {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] gas  = new int[n];
        int[] cost = new int[n];
        for (int i = 0; i < n; i++) gas[i]  = sc.nextInt();
        for (int i = 0; i < n; i++) cost[i] = sc.nextInt();

        int total = 0, tank = 0, start = 0;
        for (int i = 0; i < n; i++) {
            int diff = gas[i] - cost[i];
            total += diff;
            tank  += diff;
            if (tank < 0) { start = i + 1; tank = 0; }
        }
        if (total < 0) System.out.println("Impossible");
        else           System.out.println("Start at: " + start);
    }
}

Easy1. Fractional knapsack
Algorithm: Calculate value/weight ratio for each item. Sort descending by ratio. Fill knapsack greedily — take full item if it fits, else take fraction.
import java.util.*;

public class FractionalKnapsack {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        double W = sc.nextDouble();
        double[][] items = new double[n][2]; // [value, weight]
        for (int i = 0; i < n; i++) {
            items[i][0] = sc.nextDouble(); // value
            items[i][1] = sc.nextDouble(); // weight
        }
        // sort by ratio descending
        Arrays.sort(items, (a, b) ->
            Double.compare(b[0]/b[1], a[0]/a[1]));

        double totalValue = 0, remaining = W;
        for (double[] item : items) {
            if (remaining <= 0) break;
            double take = Math.min(item[1], remaining);
            totalValue += take * (item[0] / item[1]);
            remaining -= take;
        }
        System.out.printf("Max value: %.2f%n", totalValue);
    }
}

Med4. Maximum tasks with deadline (job sequencing)
Algorithm: Sort jobs by profit descending. For each job, find the latest free slot before its deadline (use a boolean array). Assign job to that slot. Count filled slots.
import java.util.*;

public class JobSequencing {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] jobs = new int[n][2]; // [deadline, profit]
        int maxD = 0;
        for (int i = 0; i < n; i++) {
            jobs[i][0] = sc.nextInt(); // deadline
            jobs[i][1] = sc.nextInt(); // profit
            maxD = Math.max(maxD, jobs[i][0]);
        }
        // sort by profit descending
        Arrays.sort(jobs, (a, b) -> b[1] - a[1]);

        boolean[] slot = new boolean[maxD + 1];
        int totalProfit = 0, count = 0;
        for (int[] job : jobs) {
            // find latest free slot before deadline
            for (int t = job[0]; t >= 1; t--) {
                if (!slot[t]) {
                    slot[t] = true;
                    totalProfit += job[1];
                    count++;
                    break;
                }
            }
        }
        System.out.println("Jobs done: " + count +
            ", Max profit: " + totalProfit);
    }
}
Time O(n² worst case)
Space O(maxDeadline)
